🎯 Цель урока

Настроить прямую связь между двумя ESP8266 без роутера — одна плата работает как точка доступа (AP), вторая как клиент (STA).

Это основа для создания беспроводных сенсорных сетей, пультов управления и робототехнических систем.

📡 ESP8266 #1 (Точка доступа)
      ⬇️ Wi-Fi (прямое соединение)
📡 ESP8266 #2 (Клиент)
⚠️ Важно: Для этого урока нужны две платы ESP8266, USB-кабели и компьютер с Arduino IDE.

📡 Скетч 1: ESP8266 #1 — Точка доступа (сервер)

Эта плата создаёт Wi-Fi сеть и ждёт подключения клиента.

/* ============================================================ Урок 8: ESP8266 #1 — Точка доступа (AP + сервер) ============================================================ Эта плата создаёт Wi-Fi сеть "ESP8266_Link" и ждёт подключения клиента. При получении данных — отвечает. ============================================================ */ #include <ESP8266WiFi.h> #include <ESP8266WebServer.h> // ===== НАСТРОЙКИ ТОЧКИ ДОСТУПА ===== const char* ap_ssid = "ESP8266_Link"; const char* ap_password = "12345678"; // ===== СОЗДАЁМ СЕРВЕР ===== ESP8266WebServer server(80); // ===== ОБРАБОТЧИК ЗАПРОСА ===== void handleData() { String message = server.arg("msg"); if (message == "") { server.send(200, "text/plain", "Привет! Отправь данные через ?msg=текст"); return; } // Выводим полученное сообщение в монитор порта Serial.print("📩 Получено: "); Serial.println(message); // Отправляем ответ String response = "✅ Получено: " + message; server.send(200, "text/plain", response); } // ===== ГЛАВНАЯ СТРАНИЦА ===== void handleRoot() { String html = "<!DOCTYPE html><html><head><meta charset='UTF-8'><title>ESP8266 Сервер</title></head><body>"; html += "<h1>📡 ESP8266 #1 (Сервер)</h1>"; html += "<p>Ожидаю данные от клиента...</p>"; html += "<p>Пример запроса: <code>/data?msg=Привет</code></p>"; html += "</body></html>"; server.send(200, "text/html", html); } // ===== SETUP ===== void setup() { Serial.begin(115200); delay(100); // Создаём точку доступа WiFi.softAP(ap_ssid, ap_password); Serial.println("✅ Точка доступа создана"); Serial.print("📡 SSID: "); Serial.println(ap_ssid); Serial.print("🔑 Пароль: "); Serial.println(ap_password); Serial.print("📡 IP-адрес сервера: "); Serial.println(WiFi.softAPIP()); // Настройка маршрутов server.on("/", handleRoot); server.on("/data", handleData); server.begin(); Serial.println("🚀 Сервер запущен!"); } void loop() { server.handleClient(); }

📡 Скетч 2: ESP8266 #2 — Клиент

Эта плата подключается к точке доступа и отправляет данные на сервер.

/* ============================================================ Урок 8: ESP8266 #2 — Клиент ============================================================ Эта плата подключается к точке доступа "ESP8266_Link" и отправляет данные на сервер (ESP8266 #1). ============================================================ */ #include <ESP8266WiFi.h> // ===== НАСТРОЙКИ ПОДКЛЮЧЕНИЯ ===== const char* ssid = "ESP8266_Link"; const char* password = "12345678"; // IP-адрес сервера (ESP8266 #1) const char* serverIP = "192.168.4.1"; // ===== ОТПРАВКА ДАННЫХ ===== void sendData(String message) { WiFiClient client; if (!client.connect(serverIP, 80)) { Serial.println("❌ Ошибка подключения к серверу"); return; } // Формируем HTTP-запрос String url = "/data?msg=" + message; client.print(String("GET ") + url + " HTTP/1.1\r\n"); client.print(String("Host: ") + serverIP + "\r\n"); client.print("Connection: close\r\n\r\n"); // Ждём ответ unsigned long timeout = millis() + 5000; while (client.available() == 0) { if (millis() > timeout) { Serial.println("❌ Таймаут ожидания ответа"); client.stop(); return; } } // Читаем ответ while (client.available()) { String line = client.readStringUntil('\r'); Serial.println(line); } client.stop(); Serial.println("✅ Отправлено: " + message); } // ===== SETUP ===== void setup() { Serial.begin(115200); delay(100); // Подключаемся к точке доступа Serial.print("Подключение к "); Serial.println(ssid); WiFi.begin(ssid, password); while (WiFi.status() != WL_CONNECTED) { delay(500); Serial.print("."); } Serial.println("\n✅ Подключено!"); Serial.print("📡 IP-адрес клиента: "); Serial.println(WiFi.localIP()); // Отправляем тестовое сообщение sendData("Привет с клиента!"); } void loop() { // Можно добавить периодическую отправку delay(5000); sendData("Пинг от клиента"); }

📝 Пошаговая инструкция

1 Подготовь две платы

Подключи обе ESP8266 к компьютеру через USB. Убедись, что Arduino IDE видит их (выбери правильные порты).

2 Загрузи скетч #1

В Arduino IDE выбери плату для ESP8266 #1, загрузи скетч. Открой монитор порта — увидишь IP-адрес 192.168.4.1.

3 Загрузи скетч #2

Выбери плату для ESP8266 #2, загрузи скетч. Открой монитор порта — клиент подключится к точке доступа и отправит сообщение.

4 Проверь связь

В мониторе порта ESP8266 #1 должны появиться полученные сообщения. В мониторе ESP8266 #2 — ответы от сервера.

🧪 Примеры использования

  • Беспроводной пульт — одна ESP8266 с кнопками, вторая управляет роботом
  • Датчик температуры — клиент отправляет показания, сервер выводит на дисплей
  • Удалённое управление светом — с одного ESP8266 включаешь свет на другом
  • Система "Умный дом" — несколько клиентов отправляют данные на один сервер

🤖 Дополнительные идеи

  • Передавай данные с датчика DHT22 — температура и влажность
  • Добавь шифрование — кодируй сообщения для безопасности
  • Сделай двустороннюю связь — обе платы могут отправлять и получать
  • Подключи несколько клиентов — до 8 устройств одновременно
